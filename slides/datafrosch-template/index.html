Code


import pandas as pd

df = pd.read_csv("turnout.csv")
by_region = (
    df.groupby("region")
      .agg(voters=("voter_id", "size"),
           turnout=("turned_out", "mean"))
      .sort_values("turnout", ascending=False)
)
print(by_region.head())
